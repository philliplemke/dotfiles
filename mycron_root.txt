# Tägliche Backups (daily) um 03:00 Uhr
0 3 * * * /usr/bin/rsnapshot daily >> /var/log/rsnapshot.log 2>&1

# Wöchentliche Backups (weekly) Montag um 03:30 Uhr
30 3 * * 1 /usr/bin/rsnapshot weekly >> /var/log/rsnapshot.log 2>&1

# Monatliche Backups (monthly) am 1. Tag des Monats um 04:00 Uhr
0 4 1 * * /usr/bin/rsnapshot monthly >> /var/log/rsnapshot.log 2>&1
